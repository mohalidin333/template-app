## 1. `package.json` (Add one line)
```json
{
  "scripts": {
    "postbuild": "next-sitemap"
  }
}
```

## 2. `next-sitemap.config.js` (Basic version)
```javascript
module.exports = {
  siteUrl: 'https://your-domain.com',
  generateRobotsTxt: true,
  exclude: ['/admin/*', '/api/*'],
}
```

## 3. `src/app/layout.js` (Minimal metadata)
```javascript
export const metadata = {
  title: {
    template: '%s | Your Site Name',
    default: 'Your Site Name - Your tagline',
  },
  description: 'Your site description in 150-160 characters',
  openGraph: {
    title: 'Your Site Name',
    description: 'Your site description',
    url: 'https://your-domain.com',
    siteName: 'Your Site Name',
    images: ['/og-image.jpg'],
    locale: 'en_US',
    type: 'website',
  },
  twitter: {
    card: 'summary_large_image',
    title: 'Your Site Name',
    description: 'Your site description',
    images: ['/og-image.jpg'],
  },
}

export default function RootLayout({ children }) {
  return (
    <html lang="en">
      <body>{children}</body>
    </html>
  )
}
```

## 4. `src/app/page.js` (Homepage)
```javascript
export const metadata = {
  title: 'Home',
  description: 'Specific description for homepage',
}

export default function Home() {
  return <h1>Welcome</h1>
}
```
